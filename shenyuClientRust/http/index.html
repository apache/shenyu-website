2.Step 1 Get shenyu_admin_client. (Register service need this)

#  a yml-format config file
shenyu:
  register:
    register_type: "http"
    servers: "http://127.0.0.1:9095"
    props:
      username: "admin"
      password: "123456"
fn register_test() {

    let config = ShenYuConfig::from_yaml_file("config.yml").unwrap();

}
